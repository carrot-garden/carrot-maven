008    package com.carrotgarden.maven.aws;
009    
010    import java.io.File;
011    import java.io.FileReader;
012    import java.io.FileWriter;
013    import java.io.Reader;
014    import java.io.Writer;
015    import java.util.HashMap;
016    import java.util.Map;
017    import java.util.Properties;
018    
019    import org.apache.maven.plugin.AbstractMojo;
020    import org.apache.maven.project.MavenProject;
021    import org.apache.maven.settings.Settings;
022    import org.slf4j.Logger;
023    import org.slf4j.impl.MavenLoggerFactory;
024    
025    /**
026     */
027    public abstract class CarrotAws extends AbstractMojo {
028    
029            /** do not use during class init */
030            protected Logger getLogger(final Class<?> klaz) {
031                    return MavenLoggerFactory.getLogger(klaz, getLog());
032            }
033    
034            /**
035             * @readonly
036             * @required
037             * @parameter expression="${project}"
038             */
039            protected MavenProject project;
040    
041            /**
042             * @readonly
043             * @required
044             * @parameter expression="${settings}"
045             */
046            protected Settings settings;
047    
048            //
049    
050            protected boolean isValidDirectory(final File file) {
051    
052                    if (file == null) {
053                            return false;
054                    }
055    
056                    if (!file.exists()) {
057                            return false;
058                    }
059    
060                    if (!file.isDirectory()) {
061                            return false;
062                    }
063    
064                    if (!file.canRead()) {
065                            return false;
066                    }
067    
068                    if (!file.canWrite()) {
069                            return false;
070                    }
071    
072                    return true;
073    
074            }
075    
076            protected Map<String, String> safeMap(final Properties props) {
077    
078                    final Map<String, String> map = new HashMap<String, String>();
079    
080                    if (props == null) {
081                            return map;
082                    }
083    
084                    for (final Map.Entry<Object, Object> entry : props.entrySet()) {
085                            final String key = entry.getKey().toString();
086                            final String value = entry.getValue().toString();
087                            map.put(key, value);
088                    }
089    
090                    return map;
091    
092            }
093    
094            protected Map<String, String> safeMap(final Map<String, String> props) {
095    
096                    final Map<String, String> map = new HashMap<String, String>();
097    
098                    if (props == null) {
099                            return map;
100                    }
101    
102                    return props;
103    
104            }
105    
106            protected Properties load(final File file) throws Exception {
107    
108                    final Properties props = new Properties();
109    
110                    if (file == null || !file.exists()) {
111                            getLog().debug("file == null || !file.exists()");
112                            return props;
113                    }
114    
115                    final Reader reader = new FileReader(file);
116    
117                    props.load(reader);
118    
119                    return props;
120    
121            }
122    
123            protected void save(final Properties props, final File file)
124                            throws Exception {
125    
126                    if (props == null || file == null) {
127                            getLog().debug("props == null || file == null");
128                            return;
129                    }
130    
131                    final File folder = file.getParentFile();
132    
133                    if (!folder.exists()) {
134                            folder.mkdirs();
135                    }
136    
137                    final Writer writer = new FileWriter(file);
138    
139                    props.store(writer, null);
140    
141            }
142    }
