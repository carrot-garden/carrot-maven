008package com.carrotgarden.maven.aws.cfn;
009
010import java.io.File;
011import java.util.HashMap;
012import java.util.List;
013import java.util.Map;
014import java.util.Properties;
015
016import org.apache.maven.plugin.MojoFailureException;
017
018import com.amazonaws.services.cloudformation.model.Output;
019import com.amazonaws.services.cloudformation.model.Stack;
020import com.amazonaws.services.cloudformation.model.StackStatus;
021
022/**
023 * cloud formation:
024 * 
025 * <b><a href=
026 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/APIReference/API_CreateStack.html"
027 * >create stack</a></b>
028 * 
029 * based on:
030 * 
031 * <b>stack name</b>,
032 * 
033 * <b><a href=
034 * "http://aws.amazon.com/cloudformation/aws-cloudformation-templates" >stack
035 * template</a></b>
036 * 
037 * ({@link #stackTemplateFile}),
038 * 
039 * <b><a href=
040 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/parameters-section-structure.html"
041 * >stack parameters</a></b>
042 * 
043 * ({@link #stackPropertiesInputFile} + {@link #stackInputParams}),
044 * 
045 * and produce a
046 * 
047 * <b><a href=
048 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/concept-outputs.html"
049 * >stack output</a></b>
050 * 
051 * ({@link #stackPropertiesOutputFile})
052 * 
053 * ; wait for completion or fail ({@link #stackTimeout})
054 * 
055 * 
056 * @goal cloud-formation-create
057 * 
058 * @phase prepare-package
059 * 
060 * @inheritByDefault true
061 * 
062 * @requiresDependencyResolution test
063 * 
064 */
065public class CarrotAwsCloudFormCreate extends CarrotAwsCloudForm {
066
067        /**
068         * AWS CloudFormation
069         * 
070         * <a href=
071         * "http://aws.amazon.com/cloudformation/aws-cloudformation-templates"
072         * >template</a>
073         * 
074         * file
075         * 
076         * @required
077         * @parameter default-value="./target/formation/formation.template"
078         */
079        protected File stackTemplateFile;
080
081        /**
082         * AWS CloudFormation
083         * 
084         * <a href=
085         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/parameters-section-structure.html"
086         * >Parameters Declaration</a>
087         * 
088         * stack template input parameters; optional; overrides settings from
089         * #stackPropertiesInputFile
090         * 
091         * @parameter
092         */
093        protected Map<String, String> stackInputParams = new HashMap<String, String>();
094
095        /**
096         * AWS CloudFormation stack create execution
097         * 
098         * <a href=
099         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/parameters-section-structure.html"
100         * >Parameters Declaration</a>
101         * 
102         * input properties file;
103         * 
104         * will be overridden by #stackInputParams if any
105         * 
106         * @required
107         * @parameter default-value="./target/formation/formation-input.properties"
108         */
109        protected File stackPropertiesInputFile;
110
111        /**
112         * AWS CloudFormation stack create execution result
113         * 
114         * <a href=
115         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/concept-outputs.html"
116         * >Outputs Section</a>
117         * 
118         * output properties file
119         * 
120         * @required
121         * @parameter default-value="./target/formation/formation-output.properties"
122         */
123        protected File stackPropertiesOutputFile;
124
125        /**
126         * {@inheritDoc}
127         */
128        @Override
129        public void execute() throws MojoFailureException {
130
131                try {
132
133                        getLog().info("stack create init [" + stackName + "]");
134
135                        final Properties stackInputProps = load(stackPropertiesInputFile);
136
137                        final CloudFormation formation = getCloudFormation(
138                                        stackTemplateFile, stackInputProps, stackInputParams);
139
140                        final Stack stack = formation.stackCreate();
141
142                        final StackStatus status = StackStatus.fromValue(stack
143                                        .getStackStatus());
144
145                        switch (status) {
146                        case CREATE_COMPLETE:
147                                break;
148                        default:
149                                throw new IllegalStateException("stack create failed");
150                        }
151
152                        getLog().info("stack create stack=\n" + stack);
153
154                        getLog().info("stack create output:");
155
156                        final List<Output> outputList = stack.getOutputs();
157
158                        final Properties outputProps = new Properties();
159
160                        for (final Output output : outputList) {
161
162                                final String key = output.getOutputKey();
163                                final String value = output.getOutputValue();
164
165                                outputProps.put(key, value);
166
167                                getLog().info("\t" + key + "=" + value);
168
169                        }
170
171                        save(outputProps, stackPropertiesOutputFile);
172
173                        getLog().info("stack create done [" + stackName + "]");
174
175                } catch (final Exception e) {
176
177                        throw new MojoFailureException("bada-boom", e);
178
179                }
180
181        }
182}
