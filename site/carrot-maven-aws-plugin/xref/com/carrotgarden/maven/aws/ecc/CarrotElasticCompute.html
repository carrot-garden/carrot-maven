134 	public Instance findInstance(final String instanceId) {
135 
136 		final List<String> instanceIdList = new ArrayList<String>();
137 		instanceIdList.add(instanceId);
138 
139 		final DescribeInstancesRequest request = new DescribeInstancesRequest();
140 		request.setInstanceIds(instanceIdList);
141 
142 		final DescribeInstancesResult result = amazonClient
143 				.describeInstances(request);
144 
145 		final List<Reservation> reservationList = result.getReservations();
146 
147 		switch (reservationList.size()) {
148 		case 0:
149 			return null;
150 		case 1:
151 			final Reservation reservation = reservationList.get(0);
152 			final List<Instance> instanceList = reservation.getInstances();
153 			switch (instanceList.size()) {
154 			case 0:
155 				return null;
156 			case 1:
157 				return instanceList.get(0);
158 			default:
159 				throw new IllegalStateException("duplicate instance");
160 			}
161 		default:
162 			throw new IllegalStateException("duplicate reservation");
163 		}
164 
165 	}

185 	public void instanceStart(final String instanceId) throws Exception {
186 
187 		final Instance instance = findInstance(instanceId);
188 
189 		final InstanceStateName state = stateFrom(instance);
190 
191 		logger.info("start: current state=" + state);
192 
193 		switch (state) {
194 		case Running:
195 			return;
196 		case Pending:
197 			waitForIstanceState(instanceId, InstanceStateName.Running);
198 			return;
199 		case Stopped:
200 			break;
201 		case Stopping:
202 			waitForIstanceState(instanceId, InstanceStateName.Stopped);
203 			break;
204 		case ShuttingDown:
205 		case Terminated:
206 			throw new IllegalStateException("start: dead instance");
207 		default:
208 			throw new IllegalStateException("start: unknown state");
209 		}
210 
211 		final StartInstancesRequest request = new StartInstancesRequest();
212 		request.setInstanceIds(wrapList(instanceId));
213 
214 		final StartInstancesResult result = amazonClient
215 				.startInstances(request);
216 
217 		waitForIstanceState(instanceId, InstanceStateName.Running);
218 
219 	}

259 	public Image imageCreate(final String instanceId, final String name,
260 			final String description) throws Exception {
261 
262 		logger.info("ensure instance state : instanceId=" + instanceId);
263 
264 		final InstanceStateName state = stateFrom(instanceId);
265 
266 		final boolean wasRunning;
267 
268 		switch (state) {
269 		case Pending:
270 			waitForIstanceState(instanceId, InstanceStateName.Running);
271 		case Running:
272 			wasRunning = true;
273 			break;
274 		case Stopping:
275 			waitForIstanceState(instanceId, InstanceStateName.Stopped);
276 		case Stopped:
277 			wasRunning = false;
278 			break;
279 		default:
280 			throw new Exception("image create : invalid instance state="
281 					+ state);
282 		}
283 
284 		if (wasRunning) {
285 			instanceStop(instanceId);
286 		}
287 
288 		final CreateImageRequest request = new CreateImageRequest();
289 
290 		request.setInstanceId(instanceId);
291 		request.setName(name);
292 		request.setDescription(description);
293 
294 		final CreateImageResult result = amazonClient.createImage(request);
295 
296 		final String imageId = result.getImageId();
297 
298 		logger.info("ensure image state: imageId=" + imageId);
299 
300 		final Image image = waitForImageCreate(imageId);
301 
302 		if (wasRunning) {
303 			instanceStart(instanceId);
304 		}
305 
306 		return image;
307 
308 	}

310 	/**
311 	 * @return valid image or null if missing
312 	 */
313 	public Image findImage(final String imageId) throws Exception {
314 
315 		/**
316 		 * work around for image entry not being immediately available right
317 		 * after create/register operation
318 		 */
319 		for (int index = 0; index < attemptCount; index++) {
320 
321 			try {
322 
323 				final DescribeImagesRequest request = new DescribeImagesRequest();
324 				request.setImageIds(wrapList(imageId));
325 
326 				final DescribeImagesResult result = amazonClient
327 						.describeImages(request);
328 
329 				final List<Image> imageList = result.getImages();
330 
331 				switch (imageList.size()) {
332 				case 0:
333 					logger.info("image find : missing imageId=" + imageId);
334 					break;
335 				case 1:
336 					logger.info("image find : success imageId=" + imageId);
337 					return imageList.get(0);
338 				default:
339 					logger.info("image find : duplicate imageId=" + imageId);
340 					break;
341 				}
342 
343 			} catch (final Exception e) {
344 				logger.info("image find : exception imageId={} / {}", //
345 						imageId, e.getMessage());
346 			}
347 
348 			logger.info("image find : attempt=" + index);
349 
350 			sleep();
351 
352 		}
353 
354 		logger.error("image find : failure imageId=" + imageId);
355 
356 		return null;
357 
358 	}

372 	/** delete AMI image and related EBS snapshots */
373 	public void imageDelete(final String imageId) throws Exception {
374 
375 		final Image image = findImage(imageId);
376 
377 		if (image == null) {
378 			logger.info("missing imageId = " + imageId);
379 			return;
380 		} else {
381 			logger.info("present imageId = " + imageId);
382 		}
383 
384 		imageUnregister(imageId);
385 
386 		for (final BlockDeviceMapping blockDevice : image
387 				.getBlockDeviceMappings()) {
388 
389 			final EbsBlockDevice elasticDevice = blockDevice.getEbs();
390 
391 			if (elasticDevice == null) {
392 				continue;
393 			}
394 
395 			final String snapshotId = elasticDevice.getSnapshotId();
396 
397 			if (snapshotId == null) {
398 				continue;
399 			}
400 
401 			snapshotDelete(snapshotId);
402 
403 		}
404 
405 		logger.info("removed imageId = " + imageId);
406 
407 	}

460 	private void waitForIstanceState(final String instanceId,
461 			final InstanceStateName stateName) throws Exception {
462 
463 		final long timeStart = System.currentTimeMillis();
464 
465 		while (true) {
466 
467 			final Instance instance = findInstance(instanceId);
468 
469 			if (isTimeoutPending(timeStart)) {
470 				logger.error("instance state : timeout");
471 				throw new Exception("timeout");
472 			}
473 
474 			if (instance == null) {
475 				logger.error("instance state : missing");
476 				throw new Exception("missing instance");
477 			}
478 
479 			if (stateName == stateFrom(instance)) {
480 				logger.info("instance state : done");
481 				break;
482 			} else {
483 				final long timeThis = System.currentTimeMillis();
484 				final long timeDiff = timeThis - timeStart;
485 				logger.info("instance state; time=" + timeDiff / 1000);
486 				sleep();
487 				continue;
488 			}
489 
490 		}
491 
492 	}

494 	private Image waitForImageCreate(final String imageId) throws Exception {
495 
496 		if (findImage(imageId) == null) {
497 			throw new IllegalStateException("image create: missing imageId="
498 					+ imageId);
499 		}
500 
501 		final long timeStart = System.currentTimeMillis();
502 
503 		final List<String> imageIdList = new ArrayList<String>();
504 		imageIdList.add(imageId);
505 
506 		final DescribeImagesRequest request = new DescribeImagesRequest();
507 		request.setImageIds(imageIdList);
508 
509 		while (true) {
510 
511 			final DescribeImagesResult result = amazonClient
512 					.describeImages(request);
513 
514 			final List<Image> imageList = result.getImages();
515 
516 			final Image image;
517 
518 			if (isTimeoutPending(timeStart)) {
519 				image = newImageWithStatus(ImageState.UNKNOWN.value, "timeout",
520 						"image create: timeout while waiting");
521 			} else if (imageList == null || imageList.isEmpty()) {
522 				image = newImageWithStatus(ImageState.UNKNOWN.value, "missing",
523 						"image create: missing in descriptions");
524 			} else {
525 				image = imageList.get(0);
526 			}
527 
528 			final String value = image.getState();
529 
530 			final ImageState state = ImageState.fromValue(value);
531 
532 			switch (state) {
533 
534 			case AVAILABLE:
535 				logger.info("image create: success");
536 				return image;
537 
538 			case PENDING:
539 				final long timeThis = System.currentTimeMillis();
540 				final long timeDiff = timeThis - timeStart;
541 				logger.info("image create: in progress; time=" + timeDiff
542 						/ 1000);
543 				sleep();
544 				break;
545 
546 			default:
547 				logger.error("image create: failure");
548 				return image;
549 
550 			}
551 
552 		}
