11  import org.apache.maven.plugin.MojoFailureException;

13  /**
14   * delete existing ami image with all ebs snapshots
15   * 
16   * @goal elastic-compute-image-delete
17   * 
18   * @phase prepare-package
19   * 
20   * @inheritByDefault true
21   * 
22   * @requiresDependencyResolution test
23   * 
24   */
25  public class ElastiCompImageDelete extends ElastiComp {
26  
27  	/**
28  	 * AWS ElasticCompute existing image id to delete; also can loaded via
29  	 * {@link #imageIdProperty}
30  	 * 
31  	 * @required
32  	 * @parameter default-value="ami-12345678"
33  	 */
34  	private String imageId;
35  
36  	/**
37  	 * name of project.property which, if set dynamically, will be used instead
38  	 * of static plug-in property {@link #imageId}
39  	 * 
40  	 * @parameter
41  	 */
42  	private String imageIdProperty;
43  
44  	protected String imageId() throws Exception {
45  		return projectValue(imageId, imageIdProperty);
46  	}
47  
48  	@Override
49  	public void execute() throws MojoFailureException {
50  
51  		try {
52  
53  			getLog().info("image delete init : " + imageId());
54  
55  			final CarrotElasticCompute compute = newElasticCompute();
56  
57  			compute.imageDelete(imageId());
58  
59  			getLog().info("image delete done : " + imageId());
60  
61  		} catch (final Exception e) {
62  
63  			throw new MojoFailureException("bada-boom", e);
64  
65  		}
66  
67  	}
