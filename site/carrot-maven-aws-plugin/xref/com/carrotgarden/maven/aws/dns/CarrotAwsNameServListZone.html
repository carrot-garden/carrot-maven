8   package com.carrotgarden.maven.aws.dns;
9   
10  import java.util.List;
11  
12  import org.apache.maven.plugin.MojoExecutionException;
13  import org.apache.maven.plugin.MojoFailureException;
14  
15  import com.carrotgarden.maven.aws.util.Util;
16  
17  /**
18   * route53:
19   * 
20   * <b><a href=
21   * "http://docs.amazonwebservices.com/Route53/latest/DeveloperGuide/ListingRRS.html"
22   * >list dns zone entires</a></b>
23   * 
24   * @goal route53-list-zone
25   * 
26   * @phase prepare-package
27   * 
28   * @inheritByDefault true
29   * 
30   * @requiresDependencyResolution test
31   * 
32   */
33  public class CarrotAwsNameServListZone extends CarrotAwsNameServ {
34  
35  	/**
36  	 * dns zone name to be listed
37  	 * 
38  	 * @required
39  	 * @parameter default-value="default.example.com"
40  	 */
41  	protected String dnsZoneName;
42  
43  	/**
44  	 * name of the maven project.property that will contain dns name list after
45  	 * execution of this maven goal
46  	 * 
47  	 * @required
48  	 * @parameter default-value="dnsNameList"
49  	 */
50  	protected String dnsResultProperty;
51  
52  	/**
53  	 * separator character to use when building dns result name list
54  	 * 
55  	 * @required
56  	 * @parameter default-value=";"
57  	 */
58  	protected String dnsResultSeparator;
59  
60  	@Override
61  	public void execute() throws MojoExecutionException, MojoFailureException {
62  
63  		try {
64  
65  			getLog().info("dns list init [" + dnsZoneName + "]");
66  
67  			final Route53 route53 = newRoute53();
68  
69  			final List<String> list = route53.listZone(dnsZoneName);
70  
71  			final String dnsResultNameList = Util.concatenate(list,
72  					dnsResultSeparator);
73  
74  			getLog().info("dns name list : " + dnsResultNameList);
75  
76  			project().getProperties().put(dnsResultProperty, dnsResultNameList);
77  
78  			getLog().info("dns list done [" + dnsZoneName + "]");
79  
80  		} catch (final Exception e) {
81  
82  			throw new MojoFailureException("bada-boom", e);
83  
84  		}
85  
86  	}
87  
88  }
