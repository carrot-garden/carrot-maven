008package com.carrotgarden.maven.aws.ecc;
009
010import java.io.File;
011import java.util.Properties;
012
013import org.apache.maven.plugin.MojoFailureException;
014
015import com.amazonaws.services.ec2.model.Image;
016import com.carrotgarden.maven.aws.ecc.ElasticCompute.State;
017
018/**
019 * register/create new ami image from existing AWS ElasticCompute instance
020 * 
021 * @goal elastic-compute-image-register
022 * 
023 * @phase prepare-package
024 * 
025 * @inheritByDefault true
026 * 
027 * @requiresDependencyResolution test
028 * 
029 */
030public class CarrotAwsElasCompImageReg extends CarrotAwsElasComp {
031
032        /**
033         * AWS ElasticCompute <b>instance id</b> to use for image create; if
034         * provided explicitly in pom.xml, then it will override value form
035         * properties file; this value must come either from image properties file
036         * of from pom.xml properties
037         * 
038         * @parameter
039         */
040        protected String imageInstanceId;
041
042        /**
043         * AWS ElasticCompute ami image name; must be unique under your aws account;
044         * 
045         * @required
046         * @parameter default-value="amazon-image-name"
047         */
048        protected String imageName;
049
050        /**
051         * AWS ElasticCompute ami image description
052         * 
053         * @required
054         * @parameter default-value="amazon-image-description"
055         */
056        protected String imageDescription;
057
058        /**
059         * AWS ElasticCompute image create input properties file; normally created
060         * by previous cloud formation invocation; must include InstanceId property;
061         * 
062         * @required
063         * @parameter default-value="./target/formation/formation-output.properties"
064         */
065        protected File imagePropertiesInputFile;
066
067        /**
068         * name of the property in input properties file that will be used to
069         * discover <b>InstanceId</b> to use as basis for image create operation
070         * 
071         * @required
072         * @parameter default-value="InstanceId"
073         */
074        protected String imagePropertyInstanceId;
075
076        @Override
077        public void execute() throws MojoFailureException {
078
079                try {
080
081                        getLog().info("image reg init [" + imageName + "]");
082
083                        final Properties props = load(imagePropertiesInputFile);
084
085                        final String instanceId;
086                        if (imageInstanceId == null) {
087                                instanceId = props.getProperty(imagePropertyInstanceId);
088                        } else {
089                                instanceId = imageInstanceId;
090                        }
091
092                        final ElasticCompute compute = getElasticCompute();
093
094                        final Image image = compute.imageRegister(instanceId, imageName,
095                                        imageDescription);
096
097                        final State state = State.fromValue(image.getState());
098
099                        switch (state) {
100                        case AVAILABLE:
101                                break;
102                        default:
103                                throw new IllegalStateException("image reg failed : \n" + image);
104                        }
105
106                        getLog().info("image reg done [" + imageName + "]");
107
108                } catch (final Exception e) {
109
110                        throw new MojoFailureException("bada-boom", e);
111
112                }
113
114        }
115
116}
