008    package com.carrotgarden.maven.aws.ssh;
009    
010    /**
011     */
012    
013    import java.io.File;
014    
015    import org.apache.maven.plugin.MojoFailureException;
016    
017    /**
018     * publish directory content via sftp from local to remote system
019     * 
020     * @goal secure-shell-publish
021     * 
022     * @phase prepare-package
023     * 
024     * @inheritByDefault true
025     * 
026     * @requiresDependencyResolution test
027     * 
028     */
029    public class CarrotAwsSecuShelPublish extends CarrotAwsSecuShel {
030    
031            /**
032             * ssh sftp source directory on local file system
033             * 
034             * @required
035             * @parameter default-value="./target/publish"
036             */
037            protected String sshSource;
038    
039            /**
040             * ssh sftp target directory on remote file system
041             * 
042             * @required
043             * @parameter default-value="/tmp/publish"
044             */
045            protected String sshTarget;
046    
047            /**
048             * {@inheritDoc}
049             */
050            @Override
051            public void execute() throws MojoFailureException {
052    
053                    try {
054    
055                            getLog().info("");
056    
057                            final SecureShell ssh = getSecureShell();
058    
059                            final String source = new File(sshSource).getAbsolutePath();
060                            final String target = sshTarget;
061    
062                            final int status = ssh.publish(source, target);
063    
064                            ensureStatusSuccess(status);
065    
066                    } catch (final Exception e) {
067    
068                            throw new MojoFailureException("command failed", e);
069                    }
070    
071            }
072    
073    }
