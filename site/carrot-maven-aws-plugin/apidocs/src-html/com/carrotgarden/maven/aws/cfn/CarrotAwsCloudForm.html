008    package com.carrotgarden.maven.aws.cfn;
009    
010    import java.io.File;
011    import java.util.HashMap;
012    import java.util.Map;
013    import java.util.Properties;
014    
015    import org.apache.commons.io.FileUtils;
016    import org.apache.maven.settings.Server;
017    import org.slf4j.Logger;
018    
019    import com.carrotgarden.maven.aws.CarrotAws;
020    
021    /**
022     * 
023     */
024    public abstract class CarrotAwsCloudForm extends CarrotAws {
025    
026            /**
027             * AWS CloudFormation stack name; must be unique under your aws account
028             * 
029             * @required
030             * @parameter default-value="amazon-builder"
031             */
032            protected String stackName;
033    
034            /**
035             * AWS CloudFormation
036             * 
037             * <a href=
038             * "http://docs.amazonwebservices.com/AWSSecurityCredentials/1.0/AboutAWSCredentials.html"
039             * >amazon security credentials</a>
040             * 
041             * stored in
042             * 
043             * <a href=
044             * "http://www.sonatype.com/books/mvnref-book/reference/appendix-settings-sect-details.html"
045             * >maven settings.xml</a>
046             * 
047             * under server id entry; username="Access Key ID",
048             * password="Secret Access Key";
049             * 
050             * @required
051             * @parameter default-value="com.example.aws.stack"
052             */
053            protected String stackServerId;
054    
055            /**
056             * AWS CloudFormation operation timeout; seconds
057             * 
058             * @required
059             * @parameter default-value="600"
060             */
061            protected Long stackTimeout;
062    
063            //
064    
065            protected CloudFormation getCloudFormation(final File templateFile,
066                            final Properties inputProps, final Map<String, String> inputParams)
067                            throws Exception {
068    
069                    /** merge template parameters */
070    
071                    final Map<String, String> stackParams = new HashMap<String, String>();
072    
073                    /** from properties file */
074                    stackParams.putAll(safeMap(inputProps));
075    
076                    /** from maven pom.xml */
077                    stackParams.putAll(safeMap(inputParams));
078    
079                    /** */
080    
081                    final String stackTemplate = safeTemplate(templateFile);
082    
083                    /** */
084    
085                    final Server server = settings.getServer(stackServerId);
086    
087                    if (server == null) {
088                            throw new IllegalArgumentException(
089                                            "settings.xml : server definition is missing for serverId="
090                                                            + stackServerId);
091                    }
092    
093                    final String username = server.getUsername();
094                    final String password = server.getPassword();
095    
096                    /** */
097    
098                    final Logger logger = getLogger(CloudFormation.class);
099    
100                    /** */
101    
102                    final CloudFormation formation = new CloudFormation(logger, stackName,
103                                    stackTemplate, stackParams, stackTimeout, username, password);
104    
105                    return formation;
106    
107            }
108    
109            protected String safeTemplate(final File templateFile) throws Exception {
110                    if (templateFile == null || !templateFile.exists()) {
111                            return "{}";
112                    } else {
113                            return FileUtils.readFileToString(templateFile);
114                    }
115            }
116    
117    }
